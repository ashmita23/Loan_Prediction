import re
def datee(text):
    if re.findall(r'^[\d](?=\-)',text):
        text= '0'+text
        return pd.to_datetime(text, format= '%y-%b')
    elif re.findall(r'^[\d]{2}',text):
        return pd.to_datetime(text, format= '%y-%b')
    elif re.findall(r'^[\w+]',text):
        return pd.to_datetime(text, format= '%b-%y')
